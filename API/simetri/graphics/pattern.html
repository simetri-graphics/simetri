181    @property
182    def partitions(self) -> list:
183        """
184        Returns the submatrices in the transformation.
185
186        Returns:
187            list of ndarrays.
188        """
189        if len(self.components) == 0:
190            return [identity_matrix()]
191        elif len(self.components) == 1:
192            partitions = [identity_matrix(), self.components[0].xform_matrix]
193        else:
194            partitions = []
195            for component in self.components:
196                partitions.extend(component.partitions)
197
198        return partitions

200    @property
201    def composite(self) -> 'array':
202        """
203        Returns the compound transformation matrix.
204
205        Returns:
206            ndarray: The compound transformation matrix.
207        """
208        if len(self.components) == 0:
209            return identity_matrix()
210        matrices = []
211        for component in self.components:
212            matrices.append(component.partitions)
213        res = []
214        if len(matrices) == 1:
215            if len(matrices[0]) == 1:
216                return matrices[0][0]
217            else:
218                return np.concatenate(matrices[0], axis=1)
219        else:
220            for mats in product(*matrices):
221                res.append(np.linalg.multi_dot(mats))
222
223        return np.concatenate(res, axis=1)
224
225    @composite.setter
226    def composite(self, value: 'ndarray'):
227        raise AttributeError(("Cannot set composition directly. "
228                              "Use the update method to update the composition."))

232        """
233        Creates a copy of the Transform instance.
234

236            Transform: A new Transform instance with the same components.
237        """
238        return Transformation([component.copy() for component in self.components])
