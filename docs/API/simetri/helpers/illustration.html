 434    def _update(self, xform_matrix, reps: int = 0, merge: bool = False):
 435        if reps == 0:
 436            self.xform_matrix = self.xform_matrix @ xform_matrix

 482    def text_bounds(self) -> tuple[float, float, float, float]:
 483        """Returns the bounds of the text.
 484
 485        Returns:
 486            tuple: The bounds of the text (xmin, ymin, xmax, ymax).
 487        """
 488        d_font_size = {
 489            FontSize.TINY: 5,
 490            FontSize.SMALL: 7,
 491            FontSize.NORMAL: 10,
 492            FontSize.LARGE: 12,
 493            FontSize.LARGE2: 14,
 494            FontSize.LARGE3: 17,
 495            FontSize.HUGE: 20,
 496            FontSize.HUGE2: 25,
 497        }
 498        mult = 1 # font size multiplier
 499        if self.font_size is None:
 500            font_size = defaults["font_size"]
 501        elif type(self.font_size) in [int, float]:
 502            font_size = self.font_size
 503        elif self.font_size in FontSize:
 504            font_size = convert_latex_font_size(self.font_size)
 505        else:
 506            raise ValueError("Invalid font size.")
 507        try:
 508            font = ImageFont.truetype(f"{self.font_family}.ttf", font_size)
 509        except OSError:
 510            font = ImageFont.load_default()
 511            if self.font_size in d_font_size:
 512                font_size = d_font_size[self.font_size]
 513            else:
 514                font_size = self.font_size
 515            mult = font_size / 10
 516        xmin, ymin, xmax, ymax = font.getbbox(self.text)
 517        width = xmax - xmin
 518        height = ymax - ymin
 519        dx = (width * mult) / 2
 520        dy = (height * mult) / 2
 521        xmin -= dx
 522        xmax += dx
 523        ymin -= dy
 524        ymax += dy
 525
 526        return xmin, ymin, xmax, ymax

 632def arrow(
 633    p1,
 634    p2,
 635    head_length=10,
 636    head_width=4,
 637    line_width=1,
 638    line_color=colors.black,
 639    fill_color=colors.black,
 640    centered=False,
 641):
 642    """Return an arrow from p1 to p2.
 643
 644    Args:
 645        p1 (tuple): The starting point of the arrow.
 646        p2 (tuple): The ending point of the arrow.
 647        head_length (int, optional): The length of the arrow head. Defaults to 10.
 648        head_width (int, optional): The width of the arrow head. Defaults to 4.
 649        line_width (int, optional): The width of the arrow line. Defaults to 1.
 650        line_color (Color, optional): The color of the arrow line. Defaults to colors.black.
 651        fill_color (Color, optional): The fill color of the arrow head. Defaults to colors.black.
 652        centered (bool, optional): Whether the arrow is centered. Defaults to False.
 653
 654    Returns:
 655        Batch: A Batch object containing the arrow shapes.
 656    """
 657    x1, y1 = p1[:2]
 658    x2, y2 = p2[:2]
 659    dx = x2 - x1
 660    dy = y2 - y1
 661    angle = atan2(dy, dx)
 662    body = Shape(
 663        [(x1, y1), (x2, y2)],
 664        closed=False,
 665        line_color=line_color,
 666        fill_color=fill_color,
 667        line_width=line_width,
 668    )
 669    w2 = head_width / 2
 670    head = Shape(
 671        [(-head_length, w2), (0, 0), (-head_length, -w2)],
 672        closed=True,
 673        line_color=line_color,
 674        fill_color=fill_color,
 675        line_width=line_width,
 676    )
 677    head.rotate(angle)
 678    if centered:
 679        head.translate(*mid_point((x1, y1), (x2, y2)))
 680    else:
 681        head.translate(x2, y2)
 682    return Batch([body, head])
